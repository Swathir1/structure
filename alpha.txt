#include <stdlib.h>

float real;

}complex_t;

complex_t add(complex_t, complex_t);
complex_t sub(complex_t, complex_t);
complex_t mul(complex_t, complex_t);
complex_t division(complex_t, complex_t);

complex_t add(complex_t n1, complex_t n2){
    complex_t temp;
    temp.real=n1.real+n2.real;
    temp.imag=n1.imag+n2.imag;
    return(temp);

}
complex_t sub(complex_t n1, complex_t n2){
    complex_t temp;
    temp.real=n1.real-n2.real;
    temp.imag=n1.imag-n2.imag;
    return(temp);

}
complex_t mul(complex_t n1, complex_t n2){
    complex_t temp;
    temp.real=n1.real*n2.real - n1.imag*n2.imag;
    temp.imag=n1.imag*n2.real + n1.real*n2.imag;
    return(temp);

}
